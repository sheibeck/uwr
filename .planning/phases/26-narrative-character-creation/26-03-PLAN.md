This composable manages the creation flow from the client side:

```typescript
import { computed, watch, type Ref } from 'vue';
import { tables, type DbConnection } from '../module_bindings';

// Import useTable from wherever it's imported in other composables

interface UseCharacterCreationArgs {
  connActive: Ref<boolean>;
  myIdentity: Ref<any | null>;  // Identity type from spacetimedb
}

export function useCharacterCreation({ connActive, myIdentity }: UseCharacterCreationArgs) {
  // Subscribe to creation state table
  // Use useTable if available, or direct table access pattern from existing composables
  // The key: watch CharacterCreationState rows filtered by myIdentity

  const creationState = computed(() => {
    if (!myIdentity.value) return null;
    // Get creation state rows - filter by playerId matching myIdentity
    // Use the pattern from existing composables for table access
    // tables.characterCreationState or similar
    return null; // Placeholder - implement based on actual binding pattern
  });

  const isInCreation = computed(() => {
    return creationState.value !== null && creationState.value.step !== 'COMPLETE';
  });

  const currentStep = computed(() => creationState.value?.step ?? null);

  // Watch for step changes that need procedure calls
  watch(currentStep, (newStep, oldStep) => {
    if (!connActive.value || !newStep) return;
    const conn = window.__db_conn as DbConnection | undefined;
    if (!conn) return;

    // Auto-trigger procedure when step changes to GENERATING_*
    if (newStep === 'GENERATING_RACE' && oldStep !== 'GENERATING_RACE') {
      // Call the procedure
      conn.procedures?.generateCreationContent?.({ generationType: 'race' });
      // OR if procedures aren't on conn, use the appropriate calling pattern
      // Check generated bindings for exact procedure call syntax
    }
    if (newStep === 'GENERATING_CLASS' && oldStep !== 'GENERATING_CLASS') {
      conn.procedures?.generateCreationContent?.({ generationType: 'class' });
    }
  });

  // Submit text input for creation
  function submitCreationInput(text: string) {
    if (!connActive.value) return;
    const conn = window.__db_conn as DbConnection | undefined;
    if (!conn) return;
    conn.reducers.submitCreationInput({ text });
  }

  // Start the creation flow
  function startCreation() {
    if (!connActive.value) return;
    const conn = window.__db_conn as DbConnection | undefined;
    if (!conn) return;
    conn.reducers.startCreation({});
  }

  return {
    creationState,
    isInCreation,
    currentStep,
    submitCreationInput,
    startCreation,
  };
}
```
