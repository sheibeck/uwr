export function isClassParryCapable(className: string, archetype?: string): boolean {
  const lower = className.toLowerCase();
  if (PARRY_CLASSES.has(lower)) return true;
  if (archetype === 'warrior') return true;
  return false;
}

export function isClassManaUser(className: string, archetype?: string): boolean {
  const lower = className.toLowerCase();
  if (MANA_CLASSES.has(lower) || HYBRID_MANA_CLASSES.has(lower)) return true;
  if (archetype === 'mystic') return true;
  if (archetype === 'warrior') return false;
  return false; // unknown
}

export function getBaseArmorForArchetype(archetype: string): string {

export function computeBaseStatsForGenerated(
  primaryStat: StatKey,
  secondaryStat: StatKey | undefined,
  level: bigint
): Record<StatKey, bigint> {
  const stats: Record<StatKey, bigint> = { str: BASE_STAT, dex: BASE_STAT, cha: BASE_STAT, wis: BASE_STAT, int: BASE_STAT };
  stats[primaryStat] += PRIMARY_BONUS;
  if (secondaryStat) stats[secondaryStat] += SECONDARY_BONUS;
  if (level > 1n) {
    const extraLevels = level - 1n;
    for (const key of Object.keys(stats) as StatKey[]) {
      if (key === primaryStat) stats[key] += PRIMARY_GROWTH * extraLevels;
      else if (key === secondaryStat) stats[key] += SECONDARY_GROWTH * extraLevels;
      else stats[key] += OTHER_GROWTH * extraLevels;
    }
  }
  return stats;
}

**Reducer: `submit_creation_input`** — takes `{ text: t.string() }` (no characterId — player identified by ctx.sender)

The state machine:

```
1. NO STATE EXISTS: Create CharacterCreationState row with step='AWAITING_RACE', emit greeting event via appendCreationEvent
   - Greeting: Sardonic cold open monologue. Something like: "Ah. Another one. The void spits you out and here you are, formless and fumbling, expecting me to care. I am The System. I have watched civilizations rise and crumble while you were busy not existing. But fine. Let's make something of you. Describe what manner of creature you are — your race, your people, whatever you imagine yourself to be. Be creative or be boring. I'll work with either."
   - This is NOT an LLM call — it's a hardcoded greeting to avoid burning budget on a fixed message.

2. AWAITING_RACE: Store the player's freeform text as raceDescription, set step='GENERATING_RACE'
   - The actual LLM call happens via the generate_creation_content procedure (Plan 02)
   - For now, just store the text and update step. Emit a "thinking" event: "The System is considering your... unique... heritage."

3. AWAITING_ARCHETYPE: Parse for 'warrior' or 'mystic' (case insensitive, partial match OK)
   - If recognized: store archetype, set step='GENERATING_CLASS', emit "thinking" event
   - If not recognized: emit sardonic "I offered you two choices. Warrior or Mystic. This isn't a creative writing exercise... yet." Keep step unchanged.

4. CLASS_REVEALED: Player is choosing an ability. Parse their text against the 3 ability names stored in state.abilities JSON
   - Match by case-insensitive substring of ability name
   - If matched: store chosenAbilityIndex, set step='AWAITING_NAME'
   - Emit: "A fine choice. Or a terrible one. Time will tell. Now — what shall I call you? Choose your name wisely. Or don't. I won't judge. (That's a lie.)"
   - If not matched: "I presented three abilities. Pick one by name. This isn't that complicated."

5. AWAITING_NAME: Validate name (4+ chars, no duplicates against Character table)
   - Store characterName, call finalize_character logic inline
   - Finalize: create Character row using creation state data, using computeBaseStatsForGenerated with the LLM-provided primaryStat/secondaryStat from classStats JSON, recomputeCharacterDerived, grantStarterItems, create the chosen ability in AbilityTemplate or assign it
   - Set step='COMPLETE', emit creation event: "Welcome to the world, [name]. Try not to die immediately. It's tedious to watch."
   - Also set the player's activeCharacterId

6. CONFIRMING_GO_BACK: If player typed "yes", rewind to goBackTarget step and clear data after that step. If anything else, emit "Crisis of conviction averted. Carry on." and restore previous step.
