**2. Create useSkillChoice.ts composable:**

```typescript
import { computed, watch } from 'vue';
import { useTable, tables } from '../module_bindings';

export function useSkillChoice(characterId: ComputedRef<bigint | null>) {
  const [pendingSkills, isLoading] = useTable(tables.pending_skill);

  // Filter to current character's pending skills
  const myPendingSkills = computed(() => {
    if (!characterId.value) return [];
    return pendingSkills.value.filter(
      (s: any) => s.characterId === characterId.value
    );
  });

  const hasPendingSkills = computed(() => myPendingSkills.value.length > 0);

  // Choose a skill by clicking its name
  function chooseSkill(skillName: string) {
    const skill = myPendingSkills.value.find(
      (s: any) => s.name.toLowerCase() === skillName.toLowerCase()
    );
    if (!skill) return;
    const conn = window.__db_conn;
    if (!conn) return;
    conn.reducers.chooseSkill({ pendingSkillId: skill.id });
  }

  // Trigger skill generation for a level-up
  function requestSkillGen() {
    if (!characterId.value) return;
    const conn = window.__db_conn;
    if (!conn) return;
    conn.reducers.prepareSkillGen({ characterId: characterId.value });
  }

  return {
    myPendingSkills,
    hasPendingSkills,
    isLoading,
    chooseSkill,
    requestSkillGen,
  };
}
```
