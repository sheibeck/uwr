**Step 2: Add cleanup reducer and seed** to `spacetimedb/src/index.ts`:

Register the scheduled reducer reference and define it:
```typescript
import { LlmCleanupTick } from './schema/tables';

// Register in scheduledReducers
scheduledReducers['sweep_llm_errors'] = spacetimedb.reducer('sweep_llm_errors', { arg: LlmCleanupTick.rowType }, (ctx) => {
  // Delete error requests older than 5 minutes (300 seconds = 300_000_000 microseconds)
  const cutoff = ctx.timestamp.microsSinceUnixEpoch - 300_000_000n;

  // Also delete completed requests (they should be cleaned up by the procedure, but safety net)
  for (const request of [...ctx.db.llm_request.iter()]) {
    if (request.status === 'error' && request.createdAt.microsSinceUnixEpoch < cutoff) {
      ctx.db.llm_request.id.delete(request.id);
    }
    if (request.status === 'completed' && request.createdAt.microsSinceUnixEpoch < cutoff) {
      ctx.db.llm_request.id.delete(request.id);
    }
  }
});
```

**Step 4: Create Vue composable** at `src/composables/useLlm.ts`:

```typescript
import { computed, ref, watch, type Ref } from 'vue';
import { tables, reducers } from '../module_bindings';
import { useTable } from 'spacetimedb/vue';
import type { Identity } from 'spacetimedb';

interface UseLlmArgs {
  connActive: Ref<boolean>;
  myIdentity: Ref<Identity | null>;
  selectedCharacterId: Ref<bigint | null>;
}

export function useLlm({ connActive, myIdentity, selectedCharacterId }: UseLlmArgs) {
  const pendingDomain = ref<string | null>(null);

  // Subscribe to LLM request table to monitor status
  // NOTE: LlmRequest is a private table -- check if it appears in bindings.
  // If private tables are NOT visible to clients, this composable will need to
  // rely on event_private messages instead. Adjust after inspecting bindings.
  // For now, assume the table name in bindings is `llmRequest` (camelCase of llm_request).

  // Try to use the table if it exists in bindings
  let llmRequests: any[] = [];
  let llmLoading = ref(false);

  // Check if tables.llmRequest exists in generated bindings
  // If the table is private, it won't be in the bindings and we'll need
  // to track status via the event system instead
  if ('llmRequest' in tables) {
    const [rows, loading] = useTable((tables as any).llmRequest);
    llmRequests = rows as any;
    llmLoading = loading;
  }

  // Find the active request for this player
  const activeRequest = computed(() => {
    if (!myIdentity.value || !Array.isArray(llmRequests)) return null;
    const myHex = myIdentity.value.toHexString();
    return (llmRequests as any[]).find(
      (r: any) => r.playerId?.toHexString() === myHex &&
        (r.status === 'pending' || r.status === 'processing')
    ) ?? null;
  });

  const isProcessing = computed(() => activeRequest.value !== null);

  // Request an LLM generation
  async function requestGeneration(domain: string, model: string, userPrompt: string) {
    if (!connActive.value || !selectedCharacterId.value) return;
    if (isProcessing.value) return; // Already has active request

    const conn = window.__db_conn;
    if (!conn) return;

    pendingDomain.value = domain;

    // Step 1: Call validation reducer (creates pending request)
    conn.reducers.validateLlmRequest({
      characterId: selectedCharacterId.value,
      domain,
      model,
      userPrompt,
    });
  }

  // Watch for pending requests and call the procedure
  // When validate_llm_request creates a pending row, the client sees it
  // via subscription and then calls the procedure
  watch(activeRequest, (request) => {
    if (!request || request.status !== 'pending') return;

    const conn = window.__db_conn;
    if (!conn) return;

    // Call the procedure to execute the LLM request
    // The exact API depends on generated bindings -- adjust after inspection
    if ('callLlm' in (conn as any).procedures) {
      (conn as any).procedures.callLlm({ requestId: request.id });
    } else if ('call_llm' in (conn as any).procedures) {
      (conn as any).procedures.call_llm({ requestId: request.id });
    } else {
      console.error('call_llm procedure not found in bindings. Check generated code.');
    }
  }, { immediate: true });

  return {
    requestGeneration,
    isProcessing,
    activeRequest,
    pendingDomain,
  };
}
```
