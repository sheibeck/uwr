   ```typescript
   // Composable that watches WorldGenState and auto-triggers generation procedure

   import { ref, watch, computed } from 'vue';
   import { useTable } from '../useTable';  // or wherever useTable is imported from in this project
   import { tables } from '../module_bindings';

   export function useWorldGeneration() {
     const [worldGenStates, isLoading] = useTable(tables.worldGenState);
     const isGenerating = ref(false);

     // Watch for PENDING state matching current player
     watch(worldGenStates, (states) => {
       if (!states || isGenerating.value) return;

       const myIdentity = window.__my_identity;
       if (!myIdentity) return;
       const myHex = myIdentity.toHexString();

       const pendingState = states.find(
         (s: any) => s.playerId?.toHexString?.() === myHex && s.step === 'PENDING'
       );

       if (pendingState) {
         isGenerating.value = true;
         const conn = window.__db_conn;
         if (!conn) {
           isGenerating.value = false;
           return;
         }

         conn.procedures.generateWorldRegion({ genStateId: pendingState.id })
           .then(() => { isGenerating.value = false; })
           .catch((err: any) => {
             console.error('World generation procedure failed:', err);
             isGenerating.value = false;
           });
       }
     }, { deep: true });

     // Computed: is there an active generation for the current player?
     const activeGeneration = computed(() => {
       if (!worldGenStates.value) return null;
       const myIdentity = window.__my_identity;
       if (!myIdentity) return null;
       const myHex = myIdentity.toHexString();
       return worldGenStates.value.find(
         (s: any) => s.playerId?.toHexString?.() === myHex &&
                     (s.step === 'PENDING' || s.step === 'GENERATING')
       ) || null;
     });

     return {
       isGenerating,
       activeGeneration,
       worldGenStates,
       isLoading,
     };
   }
   ```
